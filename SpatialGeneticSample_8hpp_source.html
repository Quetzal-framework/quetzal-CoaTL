42  void add(coord_type const& x, Individual const& ind) {
43  if(m_loci.empty()){
44  m_loci = ind.loci();
45  }else{
46  assert(ind.loci() == m_loci);
47  }
48  m_dictionnary[x].push_back(ind);
49  m_locations.insert(x);
50  }

52  void add(Individual const& ind, coord_type const& x) {
53  add(x, ind);
54  }

56  auto size(coord_type const& x) const {
57  assert( m_dictionnary.count(x) > 0 );
58  return m_dictionnary.at(x).size();
59  }

62  unsigned int size = 0;

69  template<typename T>
70  SpatialGeneticSample<coord_type, individual_type>& reproject(T const& projection) {
71 
72  std::map<coord_type, std::vector<individual_type>> new_dico;
73  std::set<coord_type> new_sites;
74 
75  for(auto const& it : m_dictionnary){
76  auto x = projection.reproject_to_centroid(it.first);
77  new_dico.insert(std::make_pair(x, it.second));
78  new_sites.insert(x);
79  }
80 
81  new_dico.swap(m_dictionnary);
82  new_sites.swap(m_locations);
83  return *this;
84  }

95  std::vector<individual_type> const& individuals_at(coord_type const& x) const {
96  return m_dictionnary.at(x);
97  }

99  // TODO take ploidy into account
100  auto nb_gene_copies_discarding_NA(locus_ID_type const& locus) const
101  {
102  std::map<coord_type, std::map<value_type, unsigned int>> counts;
103  for(auto const& x : get_sampling_points())
104  {
105  for(auto const& individual : individuals_at(x))
106  {
107  auto alleles = individual.alleles(locus);
108 
109  if(alleles.first.get_allelic_state() > 0)
110  {
111  counts[x][alleles.first.get_allelic_state()] += 1;
112  }
113 
114  if(alleles.second.get_allelic_state() > 0)
115  {
116  counts[x][alleles.second.get_allelic_state()] += 1;
117  }
118  }
119  }
120  return counts;
121  }
122 
123  unsigned int allelic_richness(locus_ID_type const& locus) const {
124  std::set<value_type> set;
125  auto freq = frequencies_discarding_NA(locus);
126  for(auto const& it1 : freq){
127  for(auto const& it2 : it1.second){
128  set.insert(it2.first);
129  }
130  }
131  return set.size();
132  }
133 
134  auto frequencies_discarding_NA(locus_ID_type const& locus) const {
135 
136  std::map<coord_type, std::map<value_type, double>> freq;
137 
138  auto counts = nb_gene_copies_discarding_NA(locus);
139 
140  auto get = [](auto const& a, auto const& b){return b.second;};
141 
142  for(auto const& it1 : counts)
143  {
144  double sum = std::accumulate(it1.second.begin(), it1.second.end(), 0.0, get);
145  assert(sum > 0.0);
146 
147  for(auto const& it2 : it1.second)
148  {
149  freq[it1.first][it2.first] = static_cast<double>(it2.second)/sum;
150  }
151  }
152  return freq;
153  }

172  std::set<coord_type> locations;

179  std::set<typename individual_type::locus_ID_type> extract_loci(std::map<coord_type, std::vector<individual_type>> const& data) const {
180  std::set<typename individual_type::locus_ID_type> loci = data.cbegin()->second.front().loci();
181  assert( !loci.empty());
182  for(auto const& it1 : data){
183  assert( ! it1.second.empty() );
184  for(auto const& it2 : it1.second){
185  assert(it2.loci() == loci );
186  }
187  }
188  return loci;
189  }
