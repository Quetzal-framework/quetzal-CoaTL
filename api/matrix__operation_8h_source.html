24 template<typename matrix_type>
25 auto divide_terms_by_row_sum(matrix_type const& A) {
26  using namespace boost::numeric::ublas;
27  auto v = prod(scalar_vector<double>( A.size2(), 1.0), trans(A));
28  vector<double> w ( v.size() );
29  for(unsigned int i = 0; i < v.size(); ++i){
30  double a = v (i);
31  assert( a > 0);
32  w(i) = 1.0 / a ;
33  }
34  compressed_matrix<double> S( w.size(), w.size(), 0.0 );
35  assert(S.size2() == A.size1());
36  for(unsigned int i = 0; i < v.size(); ++i ){
37  S(i,i) = w(i);
38  }
39  matrix_type M(A.size1(),A.size2());
40  axpy_prod(S, A, M);
41  return M;
42 }
