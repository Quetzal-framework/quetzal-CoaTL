144  template<typename Merger,typename Generator, typename F, typename Tree, typename U>
145  auto coalesce_along_spatial_history(forest_type<Tree> forest, F binary_op, Generator& gen, U make_tree)
146  {
147  this->check_consistency(m_history, forest);
148  auto t = m_history.nb_generations();
149  while( (forest.nb_trees() > 1) && (t > 0) )
150  {
151  may_coalesce_colocated<Merger>(forest, t, gen, binary_op, make_tree);
152  migrate_backward(forest, t, gen);
153  --t;
154  }
155  may_coalesce_colocated<Merger>(forest, t, gen, binary_op, make_tree);
156  return forest;
157  }

291  template<typename Generator, typename Forest>
292  void migrate_backward(Forest & forest, unsigned int t, Generator& gen) const noexcept
293  {
294  Forest new_forest;
295  for(auto const it : forest)
296  {
297  coord_type x = it.first;
298  new_forest.insert(m_history.backward_kernel(x, t, gen), it.second);
299  }
300  assert(forest.nb_trees() == new_forest.nb_trees());
301  forest = new_forest;
302  }

305  template<typename Merger, typename Generator, typename Tree, typename F, typename U>
306  void may_coalesce_colocated(forest_type<Tree>& forest, unsigned int t, Generator& gen, F binop, U make_tree) const noexcept
307  {
308  auto N = get_functor_N();
309  for(auto const & x : forest.positions())
310  {
311  auto range = forest.trees_at_same_position(x);
312  std::vector<Tree> v;
313  for(auto it = range.first; it != range.second; ++it)
314  {
315  v.push_back(it->second);
316  }
317  if(v.size() >= 2)
318  {
319  assert(N(x,t) >= 1 && "Population size should be positive for evaluating coalescence probability" );
320  auto last = Merger::merge(v.begin(), v.end(), N(x, t), make_tree(x,t), binop, gen );
321  forest.erase(x);
322  for(auto it = v.begin(); it != last; ++it)
323  {
324  forest.insert(x, *it);
325  }
326  }
327  }
328  }
