82  template<typename F>
83  matrix_type make_weight_matrix(std::vector<coord_type> const& coords, F w){
84  matrix_type A (coords.size(), coords.size());
85  for (unsigned i = 0; i < A.size1 (); ++ i)
86  {
87  auto x = coords.at(i);
88  for (unsigned j = 0; j < A.size2(); ++ j)
89  {
90  auto y = coords.at(j);
91  A (i, j) = w(x, y);
92  }
93  }
94  return A;
95  }
101  std::vector<distribution_type> make_distributions() const
102  {
103  std::vector<distribution_type> dists;
104  dists.reserve(m_matrix.size1());
105  for (unsigned i = 0; i < m_matrix.size1 (); ++ i)
106  {
107  std::vector<double> weights;
108  weights.reserve(m_matrix.size2());
109  for (unsigned j = 0; j < m_matrix.size2(); ++ j)
110  {
111  weights.push_back(m_matrix(i, j));
112  }
113  dists.emplace_back(weights.cbegin(), weights.cend());
114  }
115  return dists;
116  }
117  public:
121  Implementation(const Implementation&) = delete;
127  template<typename F>
128  Implementation(std::vector<point_ID_type> const& points, std::vector<coord_type> const& coords, F f):
129  m_matrix( make_weight_matrix(coords, f) ),
130  m_points(points),
131  m_coords(coords),
132  m_distributions(make_distributions())
133  {
134  }

248  template<typename F>
249  matrix_type make_weight_matrix(std::vector<coord_type> const& coords, F w)
250  {
251  matrix_type A (coords.size(), coords.size());
252  for (unsigned i = 0; i < A.size1 (); ++ i)
253  {
254  for (unsigned j = 0; j < A.size2(); ++ j)
255  {
256  auto x = coords.at(i);
257  auto y = coords.at(j);
258  A (i, j) = w(x, y);
259  }
260  }
261  return quetzal::utils::divide_terms_by_row_sum(A);
262  }
263  public:
267  Implementation(const Implementation&) = delete;
271  template<typename F>
272  Implementation(std::vector<point_ID_type> const& points, std::vector<coord_type> const& coords, F f) :
273  m_matrix( make_matrix(coords, f) ),
274  m_points(points),
275  m_coords(coords)
276  {
277  }
285  auto arrival_space(coord_type const& x) const
286  {
287  auto it = m_arrival_space_cash.find(x);
288  if(it != m_arrival_space_cash.end())
289  {
290  return it->second;
291  }
292  else
293  {
294  auto p = m_arrival_space_cash.emplace(x, retrieve_non_zero_arrival_space(x));
295  return p.first->second;
296  }
297  }
301  auto retrieve_non_zero_arrival_space(coord_type const& x) const
302  {
303  using it1_t = typename matrix_type::const_iterator1;
304  using it2_t = typename matrix_type::const_iterator2;
305  auto i = quetzal::utils::getIndexOfPointInVector(x, m_coords);
306  it1_t it1 = m_matrix.begin1();
307  std::advance(it1, i);
308  std::vector<coord_type> v;
309  v.reserve(m_matrix.size2()); // worst case
310  for (it2_t it2 = it1.begin(); it2 != it1.end(); it2++)
311  {
312  if(*it2 > 0.0) v.push_back(m_coords.at(it2.index2()));
313  }
314  return v;
315  }
319  double migration_rate(coord_type const& x, coord_type const& y)
320  {
321  return m_matrix(quetzal::utils::getIndexOfPointInVector(x, m_coords), quetzal::utils::getIndexOfPointInVector(y, m_coords) );
322  }

426  template<typename Space, typename F1, typename F2>
427  class neighboring_migration
428  {
429  private:
430  double m_emigrant_rate;
431  F1 m_friction;
432  F2 m_get_neighbors;
433  public:
434  using coord_type = Space;
435  // Constructor
436  neighboring_migration(double emigrant_rate, F1 friction, F2 get_neighbors):
437  m_emigrant_rate(emigrant_rate),
438  m_friction(friction),
439  m_get_neighbors(get_neighbors)
440  {
441  }
447  std::vector<coord_type> arrival_space(coord_type const& x) const
448  {
449  auto v = m_get_neighbors(x);
450  // let's add focal deme to have all possible arrival demes
451  v.push_back(x);
452  return v;
453  }
457  auto operator()(coord_type const& from, coord_type const& to)
458  {
459  if(from == to)
460  {
461  return 1.0 - m_emigrant_rate;
462  }
463  else
464  {
465  return m_emigrant_rate * friction_weights(from, to);
466  }
467  }
468  private:
469  double friction_weights(coord_type const& x, coord_type const& y) const
470  {
471  // TODO does it copy the whole vector?
472  auto neighbors = m_get_neighbors(x);
473  std::vector<double> v(neighbors.size(), 0.0);
474  std::transform(neighbors.begin(), neighbors.end(), v.begin(), [this](coord_type const& z) { return 1.0/(this->m_friction(z));} );
475  double sum = std::accumulate(v.begin(), v.end(), 0.0);
476  return 1.0/(m_friction(y)*sum);
477  }
478  }; // end neighboring_migration
