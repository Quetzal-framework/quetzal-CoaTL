189  template<typename F, typename G>
190  void export_to_geotiff(F f, G g, time_type const& t1, time_type const& t2, std::string const& filename) const {
191  // Number of bands to create:
192  int nBands = std::abs(t2 - t1) + 1;
193  assert(nBands >= 1);
194  // Create a model dataset with default values
195  GDALDataset* p_sink = get().GetDriver()->Create(filename.c_str(), width(), height(), nBands, GDT_Float32, NULL);
196  std::vector<float> buffer(width()*height());
197  // RasterBands begins at 1
198  GDALRasterBand* p_source_band = get().GetRasterBand(1);
199  double na = p_source_band->GetNoDataValue();
200  // Modify the defaut data in every band
201  for(int i = 1; i <= nBands; ++i ){
202  auto err = p_source_band->RasterIO( GF_Read, 0, 0, width(), height(), buffer.data(), width(), height(), GDT_Float32, 0, 0 );
203  (void)err;
204  //replace all continental cells value by 0
205  auto continental_cells = get_domain(0);
206  for(auto const& it : continental_cells)
207  {
208  buffer.at(it.x + it.y*width()) = 0.0;
209  }
210  // Replace all cells where F is defined by F
211  auto t_curr = t1 + i-1;
212  auto f_defined_cells = g(t_curr);
213  for(coord_type const& it: f_defined_cells)
214  {
215  auto xy = to_xy(it);
216  buffer.at(xy.x + (xy.y*width()) ) = static_cast<float>(f(it, t_curr));
217  }
218  //write the data
219  GDALRasterBand* p_sink_band = p_sink->GetRasterBand(i);
220  p_sink_band->SetNoDataValue(na);
221  auto err2 = p_sink_band->RasterIO( GF_Write, 0, 0, width(), height(), buffer.data(), width(), height(), GDT_Float32, 0, 0 );
222  (void)err2;
223  p_sink->FlushCache();
224  } // end modify data
225  // write metadata
226  std::vector<double> geo_transform(6);
227  get().GetGeoTransform( geo_transform.data() );
228  p_sink->SetGeoTransform( geo_transform.data() );
229  p_sink->SetProjection( get().GetProjectionRef() );
230  GDALClose( (GDALDatasetH) p_sink );
231  }

234  std::string driver_name = "ESRI Shapefile";
235  GDALDriver *poDriver;

239 
240  GDALDataset *poDS;
241  poDS = poDriver->Create( filename.c_str(), 0, 0, 0, GDT_Unknown, NULL );
242  assert( poDS != NULL );
243 
244  OGRLayer *poLayer;
245  poLayer = poDS->CreateLayer( "sample", NULL, wkbPoint, NULL );
246  assert( poLayer != NULL );
247 
248  OGRFieldDefn oField( "count", OFTInteger );
249  if( poLayer->CreateField( &oField ) != OGRERR_NONE ){
250  throw(std::string("Creating sample size field failed."));
251  }
252 
253  for(auto const& it: counts){
254  double x = it.first.lon();
255  double y = it.first.lat();
256  int sample_size = it.second;
257 
258  OGRFeature *poFeature;
259  poFeature = OGRFeature::CreateFeature( poLayer->GetLayerDefn() );
260 
261  poFeature->SetField( "count", sample_size );
262 
263  OGRPoint pt;
264  pt.setX( x );
265  pt.setY( y );
266 
267  poFeature->SetGeometry( &pt );
268 
269  if( poLayer->CreateFeature( poFeature ) != OGRERR_NONE ){
270  throw(std::string("Failed to create feature in shapefile."));
271  }
272  OGRFeature::DestroyFeature( poFeature );
273  }
274  GDALClose( poDS );
275  }

358  // Read 1 value using GDAL API. To change if access is too slow
359  double read(unsigned int x, unsigned int y, unsigned int bandID) const {
360  int nXSize = 1;
361  auto line = (float *) CPLMalloc(sizeof(float)*nXSize);
362  auto err = band(bandID).RasterIO( GF_Read, x, y, 1, 1, line, nXSize, 1, GDT_Float32, 0, 0 );
363  if(err != CE_None){
364  throw std::invalid_argument( "received negative x y or band ID" );
365  };
366  double val = double(*line);
367  CPLFree(line);
368  return(val);
369  }

372  GeographicCoordinates to_lat_lon(unsigned int x, unsigned int y) const {
373  auto lon = m_gT[1] * x + m_gT[2] * y + m_gT[1] * 0.5 + m_gT[2] * 0.5 + m_gT[0];
374  auto lat = m_gT[4] * x + m_gT[5] * y + m_gT[4] * 0.5 + m_gT[5] * 0.5 + m_gT[3];

403  bool has_same_domain_at_all_times(){
404  bool answer = true;
405  if(depth()>1){
406  std::vector<unsigned int> bands(depth()-1);
407  std::iota(bands.begin(), bands.end(), 1);
408  auto s_0 = get_domain(0);
409  auto have_same_domain = [this, &s_0](unsigned int i){
410  auto s_i = this->get_domain(i);
411  return std::equal(s_0.begin(), s_0.end(), s_i.begin());
412  };
413  answer = std::all_of(bands.begin(), bands.end(), have_same_domain);
414  }
415  return answer;
416  }
