229  std::vector<coord_type> direct_neighbors(coord_type x) const {
230  assert(is_in_spatial_extent(x));
231  x = reproject_to_centroid(x);
232  auto res = resolution();
233  std::vector<coord_type> v;
234  coord_type x1(x);
235  x1.lon() += res.lon();
236  if(is_in_spatial_extent(x1))
237  {
238  v.push_back(reproject_to_centroid(x1));
239  }
240  coord_type x2(x);
241  x2.lat() += res.lat();
242  if(is_in_spatial_extent(x2))
243  {
244  v.push_back(reproject_to_centroid(x2));
245  }
246  coord_type x3(x);
247  x3.lon() -= res.lon();
248  if(is_in_spatial_extent(x3))
249  {
250  v.push_back(reproject_to_centroid(x3));
251  }
252  coord_type x4(x);
253  x4.lat() -= res.lat();
254  if(is_in_spatial_extent(x4))
255  {
256  v.push_back(reproject_to_centroid(x4));
257  }
258  assert(v.size() >= 2);
259  assert(v.size() <= 4);
260  return v;
261  }
