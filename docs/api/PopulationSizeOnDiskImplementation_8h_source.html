133  void serialize_layer(time_type t) const
134  {
135  const std::string filename = get_archive_name(t);
136  // create and open a binary archive for output
137  std::ofstream ofs(filename);
138  // save data to archive
139  boost::archive::binary_oarchive oa(ofs);
140  // write class instance to archive
141  oa << m_populations.at(t);
142  // archive and stream closed when destructors are called, clear map
143  m_populations.erase(t);
144  }
145 
146  void deserialize_layer(time_type t) const
147  {
148  std::string filename = get_archive_name(t);
149  // create and open an archive for input
150  std::ifstream ifs(filename, std::ios::binary);
151  boost::archive::binary_iarchive ia(ifs);
152  // read class state from archive
153  std::unordered_map<Space, Value> layer;
154  ia >> layer;
155  // archive and stream closed when destructors are called
156  m_populations[t] = layer;
157  }

159  void slide_RAM_window_to(time_type t) const
160  {
161  // window is well positioned
162  if(t == m_RAM_window.first || t == m_RAM_window.second )
163  {
164  // do nothing, just read the data
165  return;
166  }
167  // windows has to go forward
168  else if( t == m_RAM_window.second + 1)
169  {
170  // we know for sure we can serialize this
171  serialize_layer( m_RAM_window.first );
172  // slide the window
173  m_RAM_window.first += 1;
174  m_RAM_window.second += 1;
175  // we don't know if we are in forward history simulation or forward reading
176  try { deserialize_layer( m_RAM_window.second);
177  } catch (const std::exception& e) {
178  //std::cout << "layer did not exist on disk, I assume it's forward simulation time" << std::endl;
179  }
180  }
181  // windows has to go backward
182  else if( t == m_RAM_window.first - 1)
183  {
184  // we know for sure we can serialize this
185  serialize_layer( m_RAM_window.second );
186  // we know for sure we should be able to deserialize this
187  deserialize_layer( m_RAM_window.first - 1 );
188  // slide back the window
189  m_RAM_window.first -= 1;
190  m_RAM_window.second -= 1;
191  }
192  // current windows is totally not aligned: assume it's (for now) only random reading access
193  else
194  {
195  // erase both layers from RAM
196  serialize_layer( m_RAM_window.first );
197  serialize_layer( m_RAM_window.second );
198  // slide the windows
199  if(t == 0)
200  {
201  m_RAM_window.first = 0;
202  m_RAM_window.second = 1;
203  }else{
204  m_RAM_window.first = t - 1;
205  m_RAM_window.second = t;
206  }
207  // read both layers from disk
208  deserialize_layer( m_RAM_window.first );
209  deserialize_layer( m_RAM_window.second );
210  }
211  }
