41  std::vector<RestrictedGrowthString> construct_all_m_blocks_partitions_of_the_set_by_algorithm_u(unsigned int nb_blocks){
42 
43  std::vector<RestrictedGrowthString> result;
44 
45  if(nb_blocks == 1){
46 
47  result.emplace_back(RGS_type(m_set.size(), 0));
48 
49  }else if(nb_blocks == m_set.size()){
50  RGS_type v(m_set.size()) ;
51  std::iota (std::begin(v), std::end(v), 0);
52  result.emplace_back(v);
53 
54  }else{
55  assert( (1 < nb_blocks) && (nb_blocks < m_set.size() ) );
56  auto n = m_set.size();
57  a = RGS_type(n+1, 0);
58  for(unsigned int j = 1; j <= nb_blocks; ++j ){ a[n - nb_blocks + j] = j -1; }
59 
60  f(nb_blocks, n, 0);
61 
62 
63  for(auto& it : m_partitions){
64  it.erase(it.begin());
65  result.emplace_back(it);
66  }
67 
68  m_partitions.clear();
69  }
70 
71 
72  return result;

85  void f(unsigned int mu, unsigned int nu, unsigned int sigma){
86  if(mu == 2){
87  visit();
88  }else{
89  f(mu-1, nu-1, (mu+sigma)%2);
90  }
91 
92  if(nu == mu+1){
93  a[mu] = mu -1;
94  visit();
95  while(a[nu] > 0){
96  a[nu] = a[nu] - 1;
97  visit();
98  }
99  }else if(nu > mu + 1){
100  if((mu + sigma)%2 == 1){
101  a[nu - 1] = mu - 1;
102  }else{
103  a[mu] = mu -1;
104  }
105 
106  if( (a[nu] + sigma) %2 == 1){
107  b(mu, nu - 1, 0);
108  }else{
109  f(mu, nu -1, 0);
110  }
111 
112  while(a[nu] > 0){
113  a[nu] = a[nu] - 1;
114  if( (a[nu] + sigma) %2 == 1){
115  b(mu, nu - 1, 0);
116  }else{
117  f(mu, nu - 1, 0);

123  void b(unsigned int mu, unsigned int nu, unsigned int sigma){
124  if(nu == mu + 1){
125  while( a[nu] < mu - 1){
126  visit();
127  a[nu] = a[nu] + 1;
128  }
129  visit();
130  a[mu] = 0;
131  }else if( nu > mu + 1){
132  if( (a[nu] + sigma) %2 == 1){
133  f(mu, nu - 1, 0);
134  }else{
135  b(mu, nu -1, 0);
136  }
137 
138  while( a[nu] < mu -1 ){
139  a[nu] = a[nu] + 1;
140  if( (a[nu] + sigma) %2 == 1){
141  f(mu, nu - 1, 0);
142  }else{
143  b(mu, nu - 1, 0);
144  }
145  }
146 
147  if( (mu + sigma) %2 == 1){
148  a[nu - 1] = 0;
149  }else{
150  a[mu] = 0;
151  }
152  }
153 
154  if(mu == 2) {
155  visit();
156  }else{
157  b(mu -1, nu-1, (mu +sigma)%2);
158  }
