79  template<typename F>
80  matrix_type make_weight_matrix(std::vector<coord_type> const& coords, F w){
81  matrix_type A (coords.size(), coords.size());
82  for (unsigned i = 0; i < A.size1 (); ++ i)
83  {
84  auto x = coords.at(i);
85  for (unsigned j = 0; j < A.size2(); ++ j)
86  {
87  auto y = coords.at(j);
88  A (i, j) = w(x, y);
89  }
90  }
91  return A;
92  }
98  std::vector<distribution_type> make_distributions() const
99  {
100  std::vector<distribution_type> dists;
101  dists.reserve(m_matrix.size1());
102  for (unsigned i = 0; i < m_matrix.size1 (); ++ i)
103  {
104  std::vector<double> weights;
105  weights.reserve(m_matrix.size2());
106  for (unsigned j = 0; j < m_matrix.size2(); ++ j)
107  {
108  weights.push_back(m_matrix(i, j));
109  }
110  dists.emplace_back(weights.cbegin(), weights.cend());
111  }
112  return dists;
113  }

118  Implementation(const Implementation&) = delete;

245  template<typename F>
246  matrix_type make_weight_matrix(std::vector<coord_type> const& coords, F w)
247  {
248  matrix_type A (coords.size(), coords.size());
249  for (unsigned i = 0; i < A.size1 (); ++ i)
250  {
251  for (unsigned j = 0; j < A.size2(); ++ j)
252  {
253  auto x = coords.at(i);
254  auto y = coords.at(j);
255  A (i, j) = w(x, y);
256  }
257  }
258  return quetzal::utils::divide_terms_by_row_sum(A);
259  }
260  public:
264  Implementation(const Implementation&) = delete;
268  template<typename F>
269  Implementation(std::vector<point_ID_type> const& points, std::vector<coord_type> const& coords, F f) :
270  m_matrix( make_matrix(coords, f) ),
271  m_points(points),
272  m_coords(coords)
273  {
274  }
282  auto arrival_space(coord_type const& x) const
283  {
284  auto it = m_arrival_space_cash.find(x);
285  if(it != m_arrival_space_cash.end())
286  {
287  return it->second;
288  }
289  else
290  {
291  auto p = m_arrival_space_cash.emplace(x, retrieve_non_zero_arrival_space(x));
292  return p.first->second;
293  }
294  }
298  auto retrieve_non_zero_arrival_space(coord_type const& x) const
299  {
300  using it1_t = typename matrix_type::const_iterator1;
301  using it2_t = typename matrix_type::const_iterator2;
302  auto i = quetzal::utils::getIndexOfPointInVector(x, m_coords);
303  it1_t it1 = m_matrix.begin1();
304  std::advance(it1, i);
305  std::vector<coord_type> v;
306  v.reserve(m_matrix.size2()); // worst case
307  for (it2_t it2 = it1.begin(); it2 != it1.end(); it2++)
308  {
309  if(*it2 > 0.0) v.push_back(m_coords.at(it2.index2()));
310  }
311  return v;
312  }
316  double migration_rate(coord_type const& x, coord_type const& y)
317  {
318  return m_matrix(quetzal::utils::getIndexOfPointInVector(x, m_coords), quetzal::utils::getIndexOfPointInVector(y, m_coords) );
319  }

423  template<typename Space, typename F1, typename F2>
424  class neighboring_migration
425  {
426  private:
427  double m_emigrant_rate;
428  F1 m_friction;
429  F2 m_get_neighbors;
430  public:
431  using coord_type = Space;
432  // Constructor
433  neighboring_migration(double emigrant_rate, F1 friction, F2 get_neighbors):
434  m_emigrant_rate(emigrant_rate),
435  m_friction(friction),
436  m_get_neighbors(get_neighbors)
437  {
438  }
444  std::vector<coord_type> arrival_space(coord_type const& x) const
445  {
446  auto v = m_get_neighbors(x);
447  // let's add focal deme to have all possible arrival demes
448  v.push_back(x);
449  return v;
450  }
454  auto operator()(coord_type const& from, coord_type const& to)
455  {
456  if(from == to)
457  {
458  return 1.0 - m_emigrant_rate;
459  }
460  else
461  {
462  return m_emigrant_rate * friction_weights(from, to);
463  }
464  }
465  private:
466  double friction_weights(coord_type const& x, coord_type const& y) const
467  {
468  // TODO does it copy the whole vector?
469  auto neighbors = m_get_neighbors(x);
470  std::vector<double> v(neighbors.size(), 0.0);
471  std::transform(neighbors.begin(), neighbors.end(), v.begin(), [this](coord_type const& z) { return 1.0/(this->m_friction(z));} );
472  double sum = std::accumulate(v.begin(), v.end(), 0.0);
473  return 1.0/(m_friction(y)*sum);
474  }
475  }; // end neighboring_migration
