38  template<typename T>
39  std::vector<coord_type> arrival_space(coord_type const& x, const T&) const
40  {
41  auto v = _landscape.four_nearest_defined_cells(x);
42  v.push_back(x);
43  return v;
44  }
45 
46  template<typename T>
47  double operator()(coord_type const& x, coord_type const& y, T const& t) const
48  {
49  auto v = arrival_space(x,t);
50  auto binop = [this, t](auto const& a, auto const& b){return a + _f(b,t);};
51  double sum = std::accumulate(v.begin(), v.end(), 0.0, binop);
52  return _f(y,t) / sum;
53  }

128  /* @brief JC69 mutational kernel
129  */
130  class JC69
131  {
132  static const std::map<char, std::string> dico;
133  public:
134  // @typedef allelic state type
135  using state_type = char;
136  /*
137  * @brief Sample a new mutational state
138  * @param state the initial mutational state
139  * @param nb_mutations the number of mutational events to simulate
140  * @param gen a random number generator
141  * @tparam Generator a random number generator
142  * @return
143  */
144  template<typename Generator>
145  static state_type mute(state_type state, unsigned int nb_mutations, Generator & gen)
146  {
147  assert( dico.find(state) != dico.end() );
148  for(unsigned int i = 0; i < nb_mutations; ++i)
149  {
150  std::uniform_int_distribution dist(0,2);
151  state = dico.at(state)[dist(gen)];
152  }
153  return state;
154  }
155  };
