210  km distanceEarth(double lat1d, double lon1d, double lat2d, double lon2d) const {
211 
212  double earthRadiusKm = 6371.0;
213  double lat1r, lon1r, lat2r, lon2r, u, v;
214 
215  lat1r = deg2rad(lat1d);
216  lon1r = deg2rad(lon1d);
217  lat2r = deg2rad(lat2d);
218  lon2r = deg2rad(lon2d);
219 
220  u = sin((lat2r - lat1r)/2);
221  v = sin((lon2r - lon1r)/2);
222 
223  km d = 2.0 * earthRadiusKm * asin(sqrt(u * u + cos(lat1r) * cos(lat2r) * v * v));
224 
225  assert(d >= 0);
226 
227  return d;
228  }
