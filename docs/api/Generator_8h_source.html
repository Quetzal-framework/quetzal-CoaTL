49  // j_max = nombre de balles dans l'urne la plus remplie.
50  template<class Callback>
51  static void f(unsigned int n, unsigned int m, unsigned int j_max, occupancy_spectrum_type& v, Callback callback){
52 
53  if(m == 0 && n == 0){
54  // probleme resolu
55  callback(std::move(v));
56  return;
57  }
58 
59  if (m != 0){
60 
61  if (n == 0) {
62  auto copy = v;
63  copy[0] = m; // ici n = 0 et m = 0, donc probleme resolu
64  callback(std::move(copy));
65  return;
66 
67  } else {
68  if( j_max > 0){
69  // vertical descent
70  for(unsigned int i = n/j_max; i >= 1; --i){
71 
72  auto copy = v;
73  copy[j_max] = i;
74 
75  if(m >= i){
76  unsigned int balls_left = n - i*j_max;
77  if(balls_left < j_max){
78  f( balls_left, m-i, balls_left, copy, callback);
79  }else {
80  f( balls_left, m-i, j_max-1, copy, callback);
81  }
82  }
83  }
84  }
85  }
86  }
87 
88  // horizontal descent
89  if( j_max == 0 && n > 0 ){
90  // unsolvable
91  return;
92  }
93 
94  auto copy = v;
95  copy[j_max] = 0;
96  f(n, m, j_max-1, copy, callback);
97 
98  }
99
