63 // String extractions, dependant on file structure
64  void analyze(std::string const& path) {
65  std::ifstream file(path);
66  std::string line;
67  unsigned int current_line = 0;
68 
69  while ( std::getline( file, line ) ){
70  if(current_line == 0){
71  m_loci_names = read_loci_names(erase_coordinates(line));
72  }else{
73  auto x = read_coordinates(line);

102  double lat = 0.;

107  while ( std::getline( iss, elem, ' ' ) && pos <= 1 ) {

111  lon = std::stod(elem);
