184  auto make_callback(unsigned int k, unsigned int N){
185  return [this, k, N](spectrum_type&& M_j){
186  auto p = compute_probability(k, N, M_j);
187  if(this->m_pred(p)){
188  this->m_support.push_back(SpectrumHandler::handle(std::move(M_j)));
189  this->m_probas.push_back(p);
190  }

194  static Int Factorial(int num) {
195  Int fact = 1;
196  for(int i=num; i>1; --i){
197  fact *= i;
198  }
199  return fact;

203  static double compute_probability(unsigned int n, unsigned int m, spectrum_type const& m_js) {
204  Float numerator(Factorial(m)*Factorial(n));
205  Float denominator(1.0);
206  int j = 0;
207  for(auto m_j : m_js){
208  denominator *= boost::multiprecision::pow(Float(Factorial(j)),Float(m_j)) * Float(Factorial(m_j));
209  ++j;
210  }
211  denominator *= Float(std::pow(m,n));
212  Float result = numerator/denominator;
213  return static_cast<double>(result) ;
214  }
