50  struct key_type
51  {
53  coord_type from;
55  coord_type to;
57  key_type() = default;
59  key_type(coord_type const& origin, coord_type const& destination):
60  from(origin),

83  struct key_hash : public std::unary_function<key_type, std::size_t>
84  {
85  std::size_t operator()(const key_type& k) const
86  {
87  size_t res = 17;
88  res = res * 31 + std::hash<coord_type>()( k.from );
89  res = res * 31 + std::hash<coord_type>()( k.to );
90  return res;
91  }
92  };
93  private:
94  mutable std::pair<Time, Time> m_RAM_window = {0,1};
96  using forward_flow_type = std::unordered_map<time_type, std::unordered_map<key_type, value_type, key_hash>>;
98  using backward_flow_type = std::unordered_map<time_type, std::unordered_map<coord_type, std::unordered_map<coord_type, value_type>>>;
100  mutable forward_flow_type m_forward_flow;
102  mutable backward_flow_type m_backward_flow;
103  public:
107  FlowOnDiskImplementation() = default;
111  value_type flux_from_to(coord_type const& from, coord_type const& to, time_type t) const
112  {
113  slide_RAM_window_to(t);
114  assert(m_forward_flow.find(t) != m_forward_flow.end());
115  assert(m_forward_flow.at(t).find(key_type(from, to)) != m_forward_flow.at(t).end());
116  return m_forward_flow.at(t).at(key_type(from,to));
117  }
122  void set_flux_from_to(coord_type const& from, coord_type const& to, time_type t, value_type v)
123  {
124  slide_RAM_window_to(t);
125  m_forward_flow[t][key_type(from, to)] = v;
126  m_backward_flow[t][to][from] = v;
127  }

152  bool flux_to_is_defined(coord_type const& to, time_type const& t) const
153  {
154  slide_RAM_window_to(t);
155  auto it1 = m_backward_flow.find(t);
156  if(it1 != m_backward_flow.end())
157  {

164  }

190  void serialize_layer(time_type t) const
191  {
192  const std::string forward_filename = get_forward_flow_archive_name(t);
193  const std::string backward_filename = get_backward_flow_archive_name(t);
194  // create and open a binary archive for output
195  std::ofstream forward_ofs(forward_filename, std::ios::binary);
196  std::ofstream backward_ofs(backward_filename, std::ios::binary);
197  // save data to archive
198  boost::archive::binary_oarchive forward_oa(forward_ofs);
199  boost::archive::binary_oarchive backward_oa(backward_ofs);
200  // write class instance to archive
201  forward_oa << m_forward_flow.at(t);
202  backward_oa << m_backward_flow.at(t);
203  // archive and stream closed when destructors are called, clear map
204  m_forward_flow.erase(t);
205  m_backward_flow.erase(t);
206  }

208  void deserialize_layer(time_type t) const
209  {
210  const std::string forward_filename = get_forward_flow_archive_name(t);
211  const std::string backward_filename = get_backward_flow_archive_name(t);
212  // create and open an archive for input
213  std::ifstream forward_ifs(forward_filename, std::ios::binary);
214  std::ifstream backward_ifs(backward_filename, std::ios::binary);
215  boost::archive::binary_iarchive forward_ia(forward_ifs);
216  boost::archive::binary_iarchive backward_ia(backward_ifs);
217  // read class state from archive
218  std::unordered_map<key_type, value_type, key_hash> forward_layer;
219  std::unordered_map<coord_type, std::unordered_map<coord_type, value_type>> backward_layer;
220  forward_ia >> forward_layer;
221  backward_ia >> backward_layer;
222  // archive and stream closed when destructors are called
223  m_forward_flow.emplace(t,forward_layer);
224  m_backward_flow.emplace(t, backward_layer);
225  }
226 
227  void slide_RAM_window_to(time_type t) const
228  {
229  // window is well positioned
230  if(t == m_RAM_window.first || t == m_RAM_window.second )
231  {
232  // do nothing, just read the data
233  return;
234  }
235  // windows has to go forward
236  else if( t == m_RAM_window.second + 1)
237  {
238  // we know for sure we can serialize this
239  serialize_layer( m_RAM_window.first );
240  // slide the window
241  m_RAM_window.first += 1;
242  m_RAM_window.second += 1;
243  // we don't know if we are in forward history simulation or forward reading
244  try { deserialize_layer( m_RAM_window.second);
245  } catch (const std::exception& e) {
246  //std::cout << "layer did not exist on disk, I assume it's forward simulation time" << std::endl;
247  }
248  }
249  // windows has to go backward
250  else if( t == m_RAM_window.first - 1)
251  {
252  // we know for sure we can serialize this
253  serialize_layer( m_RAM_window.second );
254  // we know for sure we should be able to deserialize this
255  deserialize_layer( m_RAM_window.first - 1 );
256  // slide back the window
257  m_RAM_window.first -= 1;
258  m_RAM_window.second -= 1;
259  }
260  // current windows is totally not aligned: assume it's (for now) only random reading access
261  else
262  {
263  // erase both layers from RAM
264  serialize_layer( m_RAM_window.first );
265  serialize_layer( m_RAM_window.second );
266  // slide the windows
267  if(t == 0)
268  {
269  m_RAM_window.first = 0;
270  m_RAM_window.second = 1;
271  }else{
272  m_RAM_window.first = t - 1;
273  m_RAM_window.second = t;
274  }
275 
276  // read both layers from disk
277  deserialize_layer( m_RAM_window.first );
278  deserialize_layer( m_RAM_window.second );
279  }
280  }
